    **Step 1: Add `updateSectionContent` to `src/server/routers/document.ts`.**
    Add after the existing `renameSection` mutation, following the same pattern:
    ```typescript
    updateSectionContent: requirePermission('section:manage')
      .input(z.object({
        id: z.string().uuid(),
        documentId: z.string().uuid(),
        content: z.record(z.unknown()),
      }))
      .mutation(async ({ ctx, input }) => {
        const [updated] = await db
          .update(policySections)
          .set({ content: input.content, updatedAt: new Date() })
          .where(and(
            eq(policySections.id, input.id),
            eq(policySections.documentId, input.documentId),
          ))
          .returning()

        if (!updated) {
          throw new TRPCError({ code: 'NOT_FOUND', message: 'Section not found' })
        }

        // No audit log for content edits — high-frequency saves would flood audit table.
        // Phase 6 (Versioning) creates explicit version snapshots as the auditable content events.

        return updated
      }),
    ```
