  )
}

**File 2: `app/(workspace)/policies/[id]/milestones/[milestoneId]/_components/retry-anchor-button.tsx`**

Create RetryAnchorButton per D-15, UI-SPEC:

```tsx
'use client'

import { Loader2, RefreshCw } from 'lucide-react'
import { toast } from 'sonner'
import { Button } from '@/components/ui/button'
import { trpc } from '@/src/trpc/client'

interface RetryAnchorButtonProps {
  milestoneId: string
}

export function RetryAnchorButton({ milestoneId }: RetryAnchorButtonProps) {
  const utils = trpc.useUtils()
  const retryMutation = trpc.milestone.retryAnchor.useMutation({
    onSuccess: () => {
      toast.success('Anchor retry started')
      utils.milestone.getById.invalidate({ milestoneId })
    },
    onError: () => {
      toast.error('Failed to retry anchor')
    },
  })

  return (
    <Button
      variant="outline"
      onClick={() => retryMutation.mutate({ milestoneId })}
      disabled={retryMutation.isPending}
      className="min-h-[44px]"
    >
      {retryMutation.isPending ? (
        <>
          <Loader2 className="mr-2 size-4 animate-spin" aria-hidden="true" />
          Retrying…
        </>
      ) : (
        <>
          <RefreshCw className="mr-2 size-4" aria-hidden="true" />
          Retry Anchor
        </>
      )}
    </Button>
  )
}
```

This component is:
- `variant="outline"` Button per UI-SPEC (never destructive — retry is safe)
- `min-h-[44px]` for 44px touch target accessibility per UI-SPEC
- Loading state: `Loader2 animate-spin` + "Retrying..." label, button disabled per UI-SPEC
- Default state: `RefreshCw` icon + "Retry Anchor" label per UI-SPEC copywriting contract
- Uses `trpc.milestone.retryAnchor.useMutation` from Plan 02's router extension
- Invalidates `getById` query on success to refresh milestone detail
