```typescript
export const feedbackRouter = router({
  submit: requirePermission('feedback:submit')
    .use(requireSectionAccess('sectionId'))
    .input(z.object({
      sectionId: z.string().uuid(),
      documentId: z.string().uuid(),
      feedbackType: z.enum(['issue', 'suggestion', 'endorsement', 'evidence', 'question']),
      priority: z.enum(['low', 'medium', 'high']),
      impactCategory: z.enum(['legal', 'security', 'tax', 'consumer', 'innovation', 'clarity', 'governance', 'other']),
      title: z.string().min(1).max(200),
      body: z.string().min(10).max(5000),
      suggestedChange: z.string().max(2000).optional(),
      isAnonymous: z.boolean().default(false),
    }))
    .mutation(...) // returns { id, readableId }

  list: requirePermission('feedback:read_all')
    .input(z.object({
      documentId: z.string().uuid(),
      sectionId: z.string().uuid().optional(),
      status: z.string().optional(),
      feedbackType: z.string().optional(),
      priority: z.string().optional(),
      impactCategory: z.string().optional(),
    }))
    .query(...) // returns feedback items with submitter info

  listOwn: requirePermission('feedback:read_own')
    .query(...) // returns all feedback by current user
})
```
