export const feedbackRouter = router({
  getById: requirePermission('feedback:read_own')
    .input(z.object({ id: z.string().uuid() }))
    .query(...) // returns full feedback + submitter info

  startReview: requirePermission('feedback:review')
    .input(z.object({ id: z.string().uuid() }))
    .mutation(...) // transitions to under_review

  decide: requirePermission('feedback:review')
    .input(z.object({
      id: z.string().uuid(),
      decision: z.enum(['accept', 'partially_accept', 'reject']),
      rationale: z.string().min(20).max(2000),
    }))
    .mutation(...) // transitions to accepted/partially_accepted/rejected

  close: requirePermission('feedback:review')
    .input(z.object({ id: z.string().uuid() }))
    .mutation(...) // transitions to closed
})
```

export const evidenceRouter = router({
  attach: requirePermission('evidence:upload')
    .input(z.object({
      title: z.string().min(1),
      type: z.enum(['file', 'link']),
      url: z.string().url(),
      fileName: z.string().optional(),
      fileSize: z.number().optional(),
      feedbackId: z.string().uuid().optional(),
      sectionId: z.string().uuid().optional(),
    }))
    .mutation(...)

  listByFeedback: requirePermission('evidence:read')
    .input(z.object({ feedbackId: z.string().uuid() }))
    .query(...) // returns evidence artifacts

  remove: requirePermission('evidence:upload')
    .input(z.object({ artifactId: z.string().uuid() }))
    .mutation(...)
})
```
