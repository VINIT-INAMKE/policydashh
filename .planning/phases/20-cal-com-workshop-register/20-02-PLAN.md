    ## Step 1 — `src/lib/calcom.ts`

    ```typescript
    import 'server-only'

    /**
     * cal.com API v2 client.
     *
     * WS-07. D-01 (async via Inngest), D-02 (Cal Video default location),
     * D-04 (single shared org account via CAL_API_KEY).
     *
     * Throws plain Error on 5xx (transient — Inngest should retry).
     * Throws Error with `status: 4xx` marker on 4xx (caller maps to NonRetriableError).
     */
    export interface CalEventTypeInput {
      title: string
      slug: string
      durationMinutes: number
    }

    export interface CalEventTypeCreateResult {
      id: number
    }

    export class CalApiError extends Error {
      constructor(public status: number, message: string) {
        super(message)
        this.name = 'CalApiError'
      }
    }

    export async function createCalEventType(
      input: CalEventTypeInput,
    ): Promise<CalEventTypeCreateResult> {
      const apiKey = process.env.CAL_API_KEY
      if (!apiKey) throw new CalApiError(400, 'CAL_API_KEY not set')

      const res = await fetch('https://api.cal.com/v2/event-types', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json',
          'cal-api-version': '2024-06-14',
        },
        body: JSON.stringify({
          title: input.title,
          slug: input.slug,
          // Pass BOTH field names per research OQ2
          lengthInMinutes: input.durationMinutes,
          length: input.durationMinutes,
          locations: [{ type: 'integration', integration: 'cal-video' }],
        }),
      })

      if (!res.ok) {
        const text = await res.text().catch(() => '<no body>')
        throw new CalApiError(res.status, `cal.com API ${res.status}: ${text}`)
      }

      const body = (await res.json()) as { data?: { id?: number } }
      const id = body.data?.id
      if (typeof id !== 'number') {
        throw new CalApiError(500, `cal.com returned no id: ${JSON.stringify(body)}`)
      }
      return { id }
    }
    ```

    ## Step 2 — `src/inngest/functions/workshop-created.ts`

    Mirror participate-intake.ts structure exactly. Inline triggers (Pitfall 4).

    ```typescript
    import { NonRetriableError } from 'inngest'
    import { eq } from 'drizzle-orm'
    import { inngest } from '../client'
    import { db } from '@/src/db'
    import { workshops } from '@/src/db/schema/workshops'
    import { createCalEventType, CalApiError } from '@/src/lib/calcom'

    /**
     * workshopCreatedFn — async cal.com event-type provisioning.
     *
     * Triggered by `workshop.created` emitted from the admin create mutation.
     * Calls cal.com v2 API, backfills workshops.calcomEventTypeId.
     *
     * Error policy (D-03):
     *   - cal.com 5xx  → plain Error → Inngest retries (up to 3)
     *   - cal.com 4xx  → NonRetriableError → no retry
     *   - Missing env  → NonRetriableError
     *   - Missing row  → NonRetriableError (shouldn't happen, guard anyway)
     *
     * Public listing (Plan 05) gates embed on calcomEventTypeId IS NOT NULL
     * so the workshop row persists even if this fn fails.
     */
    export const workshopCreatedFn = inngest.createFunction(
      {
        id: 'workshop-created',
        name: 'Workshop created — provision cal.com event type',
        retries: 3,
        triggers: [{ event: 'workshop.created' }],  // INLINE — Pitfall 4
      },
      async ({ event, step }) => {
        const { workshopId } = event.data as { workshopId: string; moderatorId: string }

        const workshop = await step.run('load-workshop', async () => {
          const [row] = await db.select().from(workshops).where(eq(workshops.id, workshopId)).limit(1)
          return row ?? null
        })

        if (!workshop) {
          throw new NonRetriableError(`workshop ${workshopId} not found`)
        }

        if (workshop.calcomEventTypeId) {
          return { workshopId, skipped: 'already-provisioned' }
        }

        const eventTypeId = await step.run('create-cal-event-type', async () => {
          try {
            const slug = `workshop-${workshop.id}`  // deterministic + unique
            const result = await createCalEventType({
              title: workshop.title,
              slug,
              durationMinutes: workshop.durationMinutes ?? 60,
            })
            return result.id
          } catch (err) {
            if (err instanceof CalApiError) {
              if (err.status >= 500) {
                throw err  // retry
              }
              throw new NonRetriableError(err.message)
            }
            throw new NonRetriableError(
              err instanceof Error ? err.message : String(err),
            )
          }
        })

        await step.run('backfill-calcom-event-type-id', async () => {
          await db
            .update(workshops)
            .set({ calcomEventTypeId: String(eventTypeId), updatedAt: new Date() })
            .where(eq(workshops.id, workshopId))
        })

        return { workshopId, eventTypeId, ok: true }
      },
    )
    ```
