    ```tsx
    import { verifyFeedbackToken } from '@/src/lib/feedback-token'
    import { db } from '@/src/db'
    import { and, eq } from 'drizzle-orm'
    import { workshops, workshopSectionLinks } from '@/src/db/schema/workshops'
    import { policySections } from '@/src/db/schema/documents'
    import { WorkshopFeedbackForm } from './_components/workshop-feedback-form'
    import { ExpiredLinkCard } from './_components/expired-link-card'
    // existing imports...

    export default async function ParticipatePage({
      searchParams,
    }: { searchParams: Promise<{ workshopId?: string; token?: string }> }) {
      const params = await searchParams  // Next.js 16 async searchParams
      const { workshopId, token } = params

      // Mode 1: intake (no workshopId) — render existing Phase 19 form
      if (!workshopId) {
        return <IntakePageShell />  // existing rendering
      }

      // Mode 2/3: feedback mode — validate token
      if (!token) {
        return <PageShell><ExpiredLinkCard /></PageShell>
      }

      const payload = verifyFeedbackToken(token, workshopId)
      if (!payload) {
        return <PageShell><ExpiredLinkCard /></PageShell>
      }

      // Load workshop + linked sections for the form
      const [workshop] = await db.select().from(workshops).where(eq(workshops.id, workshopId)).limit(1)
      if (!workshop) {
        return <PageShell><ExpiredLinkCard /></PageShell>
      }

      const sections = await db
        .select({ id: policySections.id, title: policySections.title })
        .from(workshopSectionLinks)
        .innerJoin(policySections, eq(policySections.id, workshopSectionLinks.sectionId))
        .where(eq(workshopSectionLinks.workshopId, workshopId))

      return (
        <div className="cl-landing min-h-screen bg-[var(--cl-surface)]">
          <main className="mx-auto max-w-2xl px-4 pt-12 pb-16 sm:px-6 sm:pt-16">
            <header className="mb-8 text-center sm:mb-12">
              <h1
                className="text-[28px] font-semibold leading-[1.2] text-[var(--cl-on-surface)]"
                style={{ fontFamily: 'var(--font-cl-headline, Newsreader, serif)' }}
              >
                Share Your Feedback
              </h1>
              <p className="mt-3 text-base leading-relaxed text-muted-foreground">
                Your insights from the workshop help shape the policy consultation.
              </p>
            </header>
            <WorkshopFeedbackForm
              workshopId={workshopId}
              token={token}
              sections={sections}
            />
          </main>
        </div>
      )
    }
    ```

    ## Step 1 — Create `app/api/intake/workshop-feedback/route.ts`

    ```typescript
    import { z } from 'zod'
    import { eq } from 'drizzle-orm'
    import { db } from '@/src/db'
    import { feedbackItems } from '@/src/db/schema/feedback'
    import {
      workshops,
      workshopFeedbackLinks,
      workshopSectionLinks,
    } from '@/src/db/schema/workshops'
    import { policySections } from '@/src/db/schema/documents'
    import { verifyFeedbackToken } from '@/src/lib/feedback-token'

    export const runtime = 'nodejs'

    const bodySchema = z.object({
      workshopId: z.string().min(1),
      token: z.string().min(1),
      rating: z.number().int().min(1).max(5),
      comment: z.string().min(1).max(4000),
      sectionId: z.string().uuid().optional(),
    })

    export async function POST(req: Request): Promise<Response> {
      let body: z.infer<typeof bodySchema>
      try {
        const json = await req.json()
        body = bodySchema.parse(json)
      } catch (err) {
        return Response.json({ error: 'Invalid request' }, { status: 400 })
      }

      // WS-15: re-verify JWT on submit (never trust client)
      const payload = verifyFeedbackToken(body.token, body.workshopId)
      if (!payload) {
        return Response.json({ error: 'Unauthorized' }, { status: 401 })
      }

      // Load workshop to get documentId (feedback requires document + section)
      const [workshop] = await db.select().from(workshops).where(eq(workshops.id, body.workshopId)).limit(1)
      if (!workshop) {
        return Response.json({ error: 'Workshop not found' }, { status: 404 })
      }

      // Resolve sectionId + documentId: user-provided or first linked section
      let sectionId = body.sectionId
      if (!sectionId) {
        const [link] = await db
          .select({ sectionId: workshopSectionLinks.sectionId })
          .from(workshopSectionLinks)
          .where(eq(workshopSectionLinks.workshopId, body.workshopId))
          .limit(1)
        sectionId = link?.sectionId
      }
      if (!sectionId) {
        return Response.json({ error: 'No section linked to this workshop' }, { status: 400 })
      }

      const [section] = await db
        .select({ id: policySections.id, documentId: policySections.documentId })
        .from(policySections)
        .where(eq(policySections.id, sectionId))
        .limit(1)
      if (!section) {
        return Response.json({ error: 'Section not found' }, { status: 404 })
      }

      // Generate readableId (mirror existing feedback pattern — e.g., FB-20-XXXX)
      const readableId = `FB-${Date.now().toString(36).toUpperCase()}`

      // Single transaction: feedbackItems + workshopFeedbackLinks
      const feedbackId = await db.transaction(async (tx) => {
        const [row] = await tx.insert(feedbackItems).values({
          readableId,
          sectionId: section.id,
          documentId: section.documentId,
          submitterId: null as unknown as string,  // public-workshop submissions allowed; schema nullable in migration if needed
          feedbackType: 'suggestion',
          priority: 'medium',
          impactCategory: 'other',
          title: `Workshop feedback (${body.rating}/5)`,
          body: body.comment,
          status: 'submitted',
          isAnonymous: true,
          source: 'workshop',  // Plan 20-01 enum
        }).returning({ id: feedbackItems.id })

        await tx.insert(workshopFeedbackLinks).values({
          workshopId: body.workshopId,
          feedbackId: row.id,
        }).onConflictDoNothing()

        return row.id
      })

      return Response.json({ ok: true, feedbackId }, { status: 200 })
    }
    ```
