    ```typescript
    /**
     * participateIntakeFn — async worker for the public /participate intake form.
     *
     * Triggered by `participate.intake` events sent from
     * app/api/intake/participate/route.ts AFTER Turnstile verification.
     *
     * Steps:
     *   1. Rate limit at run-start: 1 run per emailHash per 15 minutes (INTAKE-03).
     *      Excess events are silently dropped by Inngest (abuse prevention).
     *   2. `clerkClient().invitations.createInvitation` with `ignoreExisting: true`
     *      (INTAKE-04 + INTAKE-06 — Clerk dedup is atomic; new user = new invite,
     *      existing user = silent no-op, same response shape either way).
     *   3. `sendWelcomeEmail` — role-tailored per orgType bucket (INTAKE-05).
     *      Always runs, even when Clerk ignored the duplicate — same success
     *      response for new and existing users (no info leak).
     *
     * Error policy (Pattern 3 in 19-RESEARCH.md):
     *   - Clerk 5xx → plain Error → Inngest retries up to `retries: 3`.
     *   - Clerk 4xx → NonRetriableError → no retry, permanent failure.
     *   - Non-Clerk errors → NonRetriableError (unknown failure surface).
     *
     * Triggers MUST be inlined in createFunction options (Pitfall 4 — Inngest v4
     * type widening footgun). Do NOT extract to `const triggers = [...]`.
     */

    import { NonRetriableError } from 'inngest'
    import { clerkClient } from '@clerk/nextjs/server'
    import { isClerkAPIResponseError } from '@clerk/shared/error'
    import { inngest } from '../client'
    import { sendWelcomeEmail } from '@/src/lib/email'

    export const participateIntakeFn = inngest.createFunction(
      {
        id: 'participate-intake',
        name: 'Participate intake — Clerk invite + welcome email',
        retries: 3,
        // INTAKE-03: hard drop after 1 run per emailHash per 15m window.
        // Hash (not raw email) is the stable, privacy-preserving key.
        rateLimit: {
          key: 'event.data.emailHash',
          limit: 1,
          period: '15m',
        },
        // INLINE triggers — Pitfall 4 (Phase 16 type widening footgun).
        triggers: [{ event: 'participate.intake' }],
      },
      async ({ event, step }) => {
        const { email, name, orgType } = event.data

        // Step 1: Clerk invitation (INTAKE-04 + INTAKE-06).
        await step.run('create-clerk-invitation', async () => {
          try {
            const client = await clerkClient()
            await client.invitations.createInvitation({
              emailAddress: email,
              ignoreExisting: true,
              publicMetadata: {
                role: 'stakeholder',
                orgType,
              },
            })
          } catch (err) {
            // Clerk 5xx → retry. Clerk 4xx → permanent.
            const status =
              isClerkAPIResponseError(err) &&
              typeof (err as { status?: number }).status === 'number'
                ? (err as { status: number }).status
                : undefined
            if (status !== undefined && status >= 500) {
              // Transient — bubble up so Inngest consumes retry budget.
              throw err instanceof Error ? err : new Error(String(err))
            }
            throw new NonRetriableError(
              `Clerk invitation failed: ${err instanceof Error ? err.message : String(err)}`,
            )
          }
        })

        // Step 2: Welcome email (INTAKE-05). Fires for both new and existing
        // users — same success shape, no info leak.
        await step.run('send-welcome-email', async () => {
          await sendWelcomeEmail(email, { name, orgType, email })
        })

        return { email, orgType, ok: true }
      },
    )
    ```

    - Import `clerkClient` from `@clerk/nextjs/server` (NOT `@clerk/backend` — the tests mock the former).
    - Import `sendWelcomeEmail` from `@/src/lib/email` — the helper does not yet exist; Plan 19-03 implements it. This will create a temporary TypeScript error that resolves when 19-03 ships. This is acceptable (Wave 2 parallel plans cross-reference): tests for this file mock `@/src/lib/email` at module level so test-time imports succeed.
    - Do NOT add a try/catch around `sendWelcomeEmail` — failures should retry.
    - Do NOT rename the function ID.
    - Do NOT add `concurrency`, `throttle`, or `idempotency` options (Pitfall: `idempotency` would override `rateLimit`).
