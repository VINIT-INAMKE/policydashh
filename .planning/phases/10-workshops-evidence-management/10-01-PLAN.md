1. Create `src/server/routers/workshop.ts` with the following procedures:

   Imports: router, requirePermission from @/src/trpc/init; z from zod; db from @/src/db; all workshop schema tables from @/src/db/schema; writeAuditLog from @/src/lib/audit; ACTIONS from @/src/lib/constants; eq, gte, lt, desc, and, asc, count from drizzle-orm; evidenceArtifacts from schema; users from schema; TRPCError from @trpc/server.

   **create** — requirePermission('workshop:manage')
   - Input: { title: z.string().min(1).max(200), description: z.string().max(2000).optional(), scheduledAt: z.string().datetime(), durationMinutes: z.number().int().positive().optional(), registrationLink: z.string().url().optional() }
   - Insert into workshops table with createdBy: ctx.user.id
   - writeAuditLog with ACTIONS.WORKSHOP_CREATE, entityType 'workshop', entityId workshop.id
   - Return the created workshop

   **list** — requirePermission('workshop:read')
   - Input: { filter: z.enum(['upcoming', 'past', 'all']).default('all') }
   - Build conditions array: if 'upcoming' push gte(workshops.scheduledAt, now), if 'past' push lt(workshops.scheduledAt, now)
   - Select workshops with conditions, orderBy desc(workshops.scheduledAt)
   - Also join users table to get creatorName (users.name) for display
   - Return rows

   **getById** — requirePermission('workshop:read')
   - Input: { workshopId: z.string().uuid() }
   - Select workshop by id, join users for creatorName
   - If not found throw TRPCError NOT_FOUND
   - Also fetch linked sections (workshopSectionLinks join policySections join policyDocuments for document title), linked feedback (workshopFeedbackLinks join feedbackItems for readableId + title + status), and artifact count
   - Return workshop with sections, feedback, artifactCount

   **update** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid(), title: z.string().min(1).max(200).optional(), description: z.string().max(2000).optional(), scheduledAt: z.string().datetime().optional(), durationMinutes: z.number().int().positive().nullable().optional(), registrationLink: z.string().url().nullable().optional() }
   - Ownership check: fetch workshop, verify workshop.createdBy === ctx.user.id OR ctx.user.role === 'admin', else throw FORBIDDEN
   - Update only provided fields, set updatedAt to new Date()
   - writeAuditLog ACTIONS.WORKSHOP_UPDATE
   - Return updated workshop

   **delete** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid() }
   - Ownership check same as update
   - Delete workshop (cascades to artifacts, section links, feedback links)
   - writeAuditLog ACTIONS.WORKSHOP_DELETE
   - Return { success: true }

   **attachArtifact** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid(), title: z.string().min(1), type: z.enum(['file', 'link']), url: z.string().url(), artifactType: z.enum(['promo', 'recording', 'summary', 'attendance', 'other']), fileName: z.string().optional(), fileSize: z.number().optional() }
   - Insert into evidenceArtifacts first (uploaderId: ctx.user.id)
   - Then insert into workshopArtifacts (workshopId, artifactId, artifactType) — sequential, no transaction (Neon HTTP driver)
   - writeAuditLog ACTIONS.WORKSHOP_ARTIFACT_ATTACH
   - Return artifact

   **removeArtifact** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid(), artifactId: z.string().uuid() }
   - Delete from workshopArtifacts where workshopId and artifactId match
   - writeAuditLog ACTIONS.WORKSHOP_ARTIFACT_REMOVE
   - Return { success: true }
   - Note: do NOT delete from evidenceArtifacts — preserve the evidence record

   **listArtifacts** — requirePermission('workshop:read')
   - Input: { workshopId: z.string().uuid() }
   - Select from workshopArtifacts join evidenceArtifacts join users (for uploaderName)
   - Return artifacts with uploaderName, artifactType, all evidence fields

   **linkSection** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid(), sectionId: z.string().uuid() }
   - Insert into workshopSectionLinks (ignore conflict / use onConflictDoNothing)
   - writeAuditLog ACTIONS.WORKSHOP_SECTION_LINK
   - Return { success: true }

   **unlinkSection** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid(), sectionId: z.string().uuid() }
   - Delete from workshopSectionLinks where both match
   - writeAuditLog ACTIONS.WORKSHOP_SECTION_UNLINK
   - Return { success: true }

   **linkFeedback** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid(), feedbackId: z.string().uuid() }
   - Insert into workshopFeedbackLinks (onConflictDoNothing)
   - writeAuditLog ACTIONS.WORKSHOP_FEEDBACK_LINK
   - Return { success: true }

   **unlinkFeedback** — requirePermission('workshop:manage')
   - Input: { workshopId: z.string().uuid(), feedbackId: z.string().uuid() }
   - Delete from workshopFeedbackLinks where both match
   - writeAuditLog ACTIONS.WORKSHOP_FEEDBACK_UNLINK
   - Return { success: true }

   - Order by feedbackItems.createdAt desc
